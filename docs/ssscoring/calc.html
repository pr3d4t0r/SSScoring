 44def isValidMinimumAltitude(altitude: float) -> bool:
 45    """
 46    Reports whether an `altitude` is below the IPC and USPA valid parameters,
 47    or within `BREAKOFF_ALTITUDE` and `PERFORMACE_WINDOW_LENGTH`.  In invalid
 48    altitude doesn't invalidate a FlySight data file.  This function can be used
 49    for generating warnings.  The stock FlySightViewer scores a speed jump even
 50    if the exit was below the minimum altitude.
 51
 52    See:  FAI Competition Rules Speed Skydiving section 5.3 for details.
 53
 54    Arguments
 55    ---------
 56        altitude
 57    An altitude in meters, calculated as data.hMSL - DZ altitude.
 58
 59    Returns
 60    -------
 61    `True` if the altitude is valid.
 62    """
 63    if not isinstance(altitude, float):
 64        altitude = float(altitude)
 65    minAltitude = BREAKOFF_ALTITUDE+PERFORMANCE_WINDOW_LENGTH
 66    return altitude >= minAltitude

 95def isValidJumpISC(data: pd.DataFrame,
 96                window: PerformanceWindow) -> bool:
 97    """
 98    Validates the jump according to ISC/FAI/USPA competition rules.  A jump is
 99    valid when the speed accuracy parameter is less than 3 m/s for the whole
100    validation window duration.
101
102    Arguments
103    ---------
104        data : pd.DataFramce
105    Jumnp data in SSScoring format
106        window : ssscoring.PerformanceWindow
107    Performance window start, end values in named tuple format
108
109    Returns
110    -------
111    `True` if the jump is valid according to ISC/FAI/USPA rules.
112    """
113    if len(data) > 0:
114        accuracy = data[data.altitudeAGL <= window.validationStart].speedAccuracyISC.max()
115        return accuracy < SPEED_ACCURACY_THRESHOLD
116    else:
117        return False

139def convertFlySight2SSScoring(rawData: pd.DataFrame,
140                              altitudeDZMeters = 0.0,
141                              altitudeDZFt = 0.0):
142    """
143    Converts a raw dataframe initialized from a FlySight CSV file into the
144    SSScoring file format.  The SSScoring format uses more descriptive column
145    headers, adds the altitude in feet, and uses UNIX time instead of an ISO
146    string.
147
148    If both `altitudeDZMeters` and `altitudeDZFt` are zero then hMSL is used.
149    Otherwise, this function adjusts the effective altitude with the value.  If
150    both meters and feet values are set this throws an error.
151
152    Arguments
153    ---------
154        rawData : pd.DataFrame
155    FlySight CSV input as a dataframe
156
157        altitudeDZMeters : float
158    Drop zone height above MSL
159
160        altitudeDZFt
161    Drop zone altitudde above MSL
162
163    Returns
164    -------
165    A dataframe in SSScoring format, featuring these columns:
166
167    - timeUnix
168    - altitudeMSL
169    - altitudeAGL
170    - altitudeMSLFt
171    - altitudeAGLFt
172    - hMetersPerSecond
173    - hKMh (km/h)
174    - vMetersPerSecond
175    - vKMh (km/h)
176    - angle
177    - speedAccuracy (ignore; see ISC documentation)
178    - hMetersPerSecond
179    - hKMh
180    - latitude
181    - longitude
182    - verticalAccuracy
183    - speedAccuracyISC
184
185    Errors
186    ------
187    `SSScoringError` if the DZ altitude is set in both meters and feet.
188    """
189    if not isinstance(rawData, pd.DataFrame):
190        raise SSScoringError('convertFlySight2SSScoring input must be a FlySight CSV dataframe')
191
192    if altitudeDZMeters and altitudeDZFt:
193        raise SSScoringError('Cannot set altitude in meters and feet; pick one')
194
195    if altitudeDZMeters:
196        altitudeDZFt = FT_IN_M*altitudeDZMeters
197    if altitudeDZFt:
198        altitudeDZMeters = altitudeDZFt/FT_IN_M
199
200    data = rawData.copy()
201
202    data['altitudeMSLFt'] = data['hMSL'].apply(lambda h: FT_IN_M*h)
203    data['altitudeAGL'] = data.hMSL-altitudeDZMeters
204    data['altitudeAGLFt'] = data.altitudeMSLFt-altitudeDZFt
205    data['timeUnix'] = np.round(data['time'].apply(lambda t: pd.Timestamp(t).timestamp()), decimals = 2)
206    data['hMetersPerSecond'] = (data.velE**2.0+data.velN**2.0)**0.5
207    speedAngle = data['hMetersPerSecond']/data['velD']
208    speedAngle = np.round(90.0-speedAngle.apply(math.atan)/DEG_IN_RADIANS, decimals = 2)
209    speedAccuracyISC = np.round(data.vAcc.apply(lambda a: (2.0**0.5)*a/3.0), decimals = 2)
210
211    data = pd.DataFrame(data = {
212        'timeUnix': data.timeUnix,
213        'altitudeMSL': data.hMSL,
214        'altitudeAGL': data.altitudeAGL,
215        'altitudeMSLFt': data.altitudeMSLFt,
216        'altitudeAGLFt': data.altitudeAGLFt,
217        'vMetersPerSecond': data.velD,
218        'vKMh': 3.6*data.velD,
219        'speedAngle': speedAngle,
220        'speedAccuracy': data.sAcc,
221        'hMetersPerSecond': data.hMetersPerSecond,
222        'hKMh': 3.6*data.hMetersPerSecond,
223        'latitude': data.lat,
224        'longitude': data.lon,
225        'verticalAccuracy': data.vAcc,
226        'speedAccuracyISC': speedAccuracyISC,
227    })
228
229    return data

232def _dataGroups(data):
233    data_ = data.copy()
234    data_['positive'] = (data_.vMetersPerSecond > 0)
235    data_['group'] = (data_.positive != data_.positive.shift(1)).astype(int).cumsum()-1

240def getSpeedSkydiveFrom(data: pd.DataFrame) -> tuple:
241    """
242    Take the skydive dataframe and get the speed skydiving data:
243
244    - Exit
245    - Speed skydiving window
246    - Drops data before exit and below breakoff altitude
247
248    Arguments
249    ---------
250        data : pd.DataFrame
251    Jump data in SSScoring format
252
253    Returns
254    -------
255    A tuple of two elements:
256
257    - A named tuple with performance and validation window data
258    - A dataframe featuring only speed skydiving data
259
260    Warm up FlySight files and non-speed skydiving files may return invalid
261    values:
262
263    - `None` for the `PerformanceWindow` instance
264    - `data`, most likely empty
265    """
266    if len(data):
267        data = _dataGroups(data)
268        groups = data.group.max()+1
269
270        freeFallGroup = -1
271        MIN_DATA_POINTS = 100 # heuristic
272        MIN_MAX_SPEED = 200 # km/h, heuristic; slower ::= no free fall
273        for group in range(groups):
274            subset = data[data.group == group]
275            if len(subset) >= MIN_DATA_POINTS and subset.vKMh.max() >= MIN_MAX_SPEED:
276                freeFallGroup = group
277
278        data = data[data.group == freeFallGroup]
279        data = data.drop('group', axis = 1).drop('positive', axis = 1)
280
281    if len(data) > 0:
282        exitTime = data[data.vMetersPerSecond > EXIT_SPEED].head(1).timeUnix.iat[0]
283        data = data[data.timeUnix >= exitTime]
284        data = data[data.altitudeAGL >= BREAKOFF_ALTITUDE]
285
286        windowStart = data.iloc[0].altitudeAGL
287        windowEnd = windowStart-PERFORMANCE_WINDOW_LENGTH
288        if windowEnd < BREAKOFF_ALTITUDE:
289            windowEnd = BREAKOFF_ALTITUDE
290
291        validationWindowStart = windowEnd+VALIDATION_WINDOW_LENGTH
292        data = data[data.altitudeAGL >= windowEnd]
293
294        return PerformanceWindow(windowStart, windowEnd, validationWindowStart), data
295    else:
296        return None, data

304def jumpAnalysisTable(data: pd.DataFrame) -> pd.DataFrame:
305    """
306    Generates the HCD jump analysis table, with speed data at 5-second intervals
307    after exit.
308
309    Arguments
310    ---------
311        data : pd.DataFrame
312    Jump data in SSScoring format
313
314    Returns
315    -------
316    A tuple with a pd.DataFrame and the max speed recorded for the jump:
317
318    - A table dataframe with time and speed
319    - a floating point number
320    """
321    table = None
322    distanceStart = (data.iloc[0].latitude, data.iloc[0].longitude)
323    for column in pd.Series([ 5.0, 10.0, 15.0, 20.0, 25.0, ]):
324        for interval in range(int(column)*10, 10*(int(column)+1)):
325            # Use the next 0.1 sec interval if the current interval tranche has
326            # NaN values.
327            columnRef = interval/10.0
328            timeOffset = data.iloc[0].timeUnix+columnRef
329            tranche = data.query('timeUnix == %f' % timeOffset).copy()
330            tranche['time'] = [ column, ]
331            currentPosition = (tranche.iloc[0].latitude, tranche.iloc[0].longitude)
332            tranche['distanceFromExit'] = [ round(calculateDistance(distanceStart, currentPosition), 2), ]
333            if not tranche.isnull().any().any():
334                break
335
336        if pd.isna(tranche.iloc[-1].vKMh):
337            tranche = data.tail(1).copy()
338            currentPosition = (tranche.iloc[0].latitude, tranche.iloc[0].longitude)
339            tranche['time'] = tranche.timeUnix-data.iloc[0].timeUnix
340            tranche['distanceFromExit'] = [ round(calculateDistance(distanceStart, currentPosition), 2), ]
341
342        if table is not None:
343            table = pd.concat([ table, tranche, ])
344        else:
345            table = tranche
346    table = pd.DataFrame({
347                'time': table.time,
348                'vKMh': table.vKMh,
349                'deltaV': table.vKMh.diff().fillna(table.vKMh),
350                'vAccel m/s²': _verticalAcceleration(table.vKMh, table.time),
351                'speedAngle': table.speedAngle,
352                'angularVel º/s': (table.speedAngle.diff()/table.time.diff()).fillna(table.speedAngle/TABLE_INTERVAL),
353                'deltaAngle': table.speedAngle.diff().fillna(table.speedAngle),
354                'hKMh': table.hKMh,
355                'distanceFromExit (m)': table.distanceFromExit,
356                'altitude (ft)': table.altitudeAGLFt,
357            })
358    return table

361def dropNonSkydiveDataFrom(data: pd.DataFrame) -> pd.DataFrame:
362    """
363    Discards all data rows before maximum altitude, and all "negative" altitude
364    rows because we don't skydive underground (FlySight bug?).
365
366    This is a more accurate mean velocity calculation from a physics point of
367    view, but it differs from the ISC definition using in scoring - which, if we
368    get technical, is the one that counts.
369
370    Arguments
371    ---------
372        data : pd.DataFrame
373    Jump data in SSScoring format (headers differ from FlySight format)
374
375    Returns
376    -------
377    The jump data for the skydive
378    """
379    timeMaxAlt = data[data.altitudeAGL == data.altitudeAGL.max()].timeUnix.iloc[0]
380    data = data[data.timeUnix > timeMaxAlt]
381
382    data = data[data.altitudeAGL > 0]
383
384    return data

387def calcScoreMeanVelocity(data: pd.DataFrame) -> tuple:
388    """
389    Calculates the speeds over a 3-second interval as the mean of all the speeds
390    recorded within that 3-second window and resolves the maximum speed.
391
392    Arguments
393    ---------
394        data
395    A `pd.dataframe` with speed run data.
396
397    Returns
398    -------
399    A `tuple` with the best score throughout the speed run, and a dicitionary
400    of the meanVSpeed:spotInTime used in determining the exact scoring speed
401    at every datat point during the speed run.
402
403    Notes
404    -----
405    This implementation uses iteration instead of binning/factorization because
406    some implementers may be unfamiliar with data manipulation in dataframes
407    and this is a critical function that may be under heavy review.  Future
408    versions may revert to dataframe and series/np.array factorization.
409    """
410    scores = dict()
411    for spot in data.plotTime[::1]:
412        subset = data[(data.plotTime <= spot) & (data.plotTime >= (spot-SCORING_INTERVAL))]
413        scores[np.round(subset.vKMh.mean(), decimals = 2)] = spot
414    return (max(scores), scores)

417def calcScoreISC(data: pd.DataFrame) -> tuple:
418    """
419    Calculates the speeds over a 3-second interval as the ds/dt and dt is the
420    is a 3-second sliding interval from exit.  The window slider moves along the
421    `plotTime` axis in the dataframe.
422
423    Arguments
424    ---------
425        data
426    A `pd.dataframe` with speed run data.
427
428    Returns
429    -------
430    A `tuple` with the best score throughout the speed run, and a dicitionary
431    of the meanVSpeed:spotInTime used in determining the exact scoring speed
432    at every datat point during the speed run.
433    """
434    scores = dict()
435    step = data.plotTime.diff().dropna().mode().iloc[0]
436    end = data.plotTime[-1:].iloc[0]-SCORING_INTERVAL
437    for spot in np.arange(0.0, end, step):
438        intervalStart = np.round(spot, decimals = 2)
439        intervalEnd = np.round(intervalStart+SCORING_INTERVAL, decimals = 2)
440        try:
441            h1 = data[data.plotTime == intervalStart].altitudeAGL.iloc[0]
442            h2 = data[data.plotTime == intervalEnd].altitudeAGL.iloc[0]
443        except IndexError:
444            # TODO: Decide whether to log the missing FlySight samples.
445            continue
446        intervalScore = np.round(MPS_2_KMH*abs(h1-h2)/SCORING_INTERVAL, decimals = 2)
447        scores[intervalScore] = intervalStart
448    return (max(scores), scores)

451def processJump(data: pd.DataFrame) -> JumpResults:
452    """
453    Take a dataframe in SSScoring format and process it for display.  It
454    serializes all the steps that would be taken from the ssscoring module, but
455    includes some text/HTML data in the output.
456
457    Arguments
458    ---------
459        data: pd.DataFrame
460    A dataframe in SSScoring format
461
462    Returns
463    -------
464    A `JumpResults` named tuple with these items:
465
466    - `score` speed score
467    - `maxSpeed` maximum speed during the jump
468    - `scores` a Series of every 3-second window scores from exit to breakoff
469    - `data` an updated SSScoring dataframe `plotTime`, where 0 = exit, used
470      for plotting
471    - `window` a named tuple with the exit, breakoff, and validation window
472      altitudes
473    - `table` a dataframe featuring the speeds and altitudes at 5-sec intervals
474    - `color` a string that defines the color for the jump result; possible
475      values are _green_ for valid jump, _red_ for invalid jump, per ISC rules
476    - `result` a string with the legend of _valid_ or _invalid_ jump
477    """
478    workData = data.copy()
479    workData = dropNonSkydiveDataFrom(workData)
480    window, workData = getSpeedSkydiveFrom(workData)
481    if workData.empty and not window:
482        workData = None
483        maxSpeed = -1.0
484        score = -1.0
485        scores = None
486        table = None
487        window = None
488        jumpStatus = JumpStatus.WARM_UP_FILE
489    else:
490        validJump = isValidJumpISC(workData, window)
491        jumpStatus = JumpStatus.OK
492        score = None
493        scores = None
494        table = None
495        if validJump:
496            table = None
497            table = jumpAnalysisTable(workData)
498            maxSpeed = data.vKMh.max()
499            baseTime = workData.iloc[0].timeUnix
500            workData['plotTime'] = round(workData.timeUnix-baseTime, 2)
501            score, scores = calcScoreISC(workData)
502        else:
503            maxSpeed = -1
504            if len(workData):
505                jumpStatus = JumpStatus.SPEED_ACCURACY_EXCEEDS_LIMIT
506            else:
507                jumpStatus = JumpStatus.INVALID_SPEED_FILE
508    return JumpResults(workData, maxSpeed, score, scores, table, window, jumpStatus)

537def getFlySightDataFromCSVBuffer(buffer:bytes, bufferName:str) -> tuple:
538    """
539    Ingress a buffer with known FlySight or SkyTrax file data for SSScoring
540    processing.
541
542    Arguments
543    ---------
544        buffer
545    A binary data buffer, bag of bytes, containing a known FlySight track file.
546
547        bufferName
548    An arbitrary name for the buffer of type `str`.  It's used for constructing
549    the full buffer tag value for human identification.
550
551    Returns
552    -------
553    A `tuple` with two items:
554        - `rawData` - a dataframe representation of the CSV with the original
555          headers but without the data type header
556        - `tag` - a string with an identifying tag derived from the path name
557          and file version in the form `some name:vX`.  It uses the current
558          path as metadata to infer the name.  There's no semantics enforcement.
559
560    Raises
561    ------
562    `SSScoringError` if the CSV file is invalid in any way.
563    """
564    if not isinstance(buffer, bytes):
565        raise SSScoringError('buffer must be an instance of bytes, a bytes buffer')
566    try:
567        stringIO = StringIO(buffer.decode(FLYSIGHT_FILE_ENCODING))
568    except Exception as e:
569        raise SSScoringError('invalid buffer endcoding - %s' % str(e))
570    version = detectFlySightFileVersionOf(buffer)
571    if version == FlySightVersion.V1:
572        rawData = _readVersion1CSV(stringIO)
573        tag = _tagVersion1From(bufferName)
574    elif version == FlySightVersion.V2:
575        rawData = _readVersion2CSV(stringIO)
576        tag = _tagVersion2From(bufferName)
577    return (rawData, tag)

580def getFlySightDataFromCSVFileName(jumpFile) -> tuple:
581    """
582    Ingress a known FlySight or SkyTrax file into memory for SSScoring
583    processing.
584
585    Arguments
586    ---------
587        jumpFile
588    A string or `pathlib.Path` object; can be a relative or an asbolute path.
589
590    Returns
591    -------
592    A `tuple` with two items:
593        - `rawData` - a dataframe representation of the CSV with the original
594          headers but without the data type header
595        - `tag` - a string with an identifying tag derived from the path name
596          and file version in the form `some name:vX`.  It uses the current
597          path as metadata to infer the name.  There's no semantics enforcement.
598
599    Raises
600    ------
601    `SSScoringError` if the CSV file is invalid in any way.
602    """
603    from ssscoring.flysight import validFlySightHeaderIn
604
605    if isinstance(jumpFile, Path):
606        jumpFile = jumpFile.as_posix()
607    elif isinstance(jumpFile, str):
608        pass
609    else:
610        raise SSScoringError('jumpFile must be a string or a Path object')
611    if not validFlySightHeaderIn(jumpFile):
612        raise SSScoringError('%s is an invalid speed skydiving file')
613    version = detectFlySightFileVersionOf(jumpFile)
614    if version == FlySightVersion.V1:
615        rawData = _readVersion1CSV(jumpFile)
616        tag = _tagVersion1From(jumpFile)
617    elif version == FlySightVersion.V2:
618        rawData = _readVersion2CSV(jumpFile)
619        tag = _tagVersion2From(jumpFile)
620    return (rawData, tag)
