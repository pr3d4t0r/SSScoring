 64def isStreamlitHostedApp() -> bool:
 65    """
 66    Detect if the hosting environment is a native Python system or a Streamlit
 67    app environment hosted by streamlit.io or Snowflake.
 68
 69    Returns
 70    -------
 71    `True` if the app is running in the Streamlit app or Snoflake app
 72    environment, otherwise `False`.
 73    """
 74    keys = tuple(os.environ.keys())
 75    if STREAMLIT_SIG_KEY not in keys:
 76        return False
 77    if os.environ[STREAMLIT_SIG_KEY] == STREAMLIT_SIG_VALUE:
 78        return True
 79    return False

 82@st.cache_data
 83def fetchResource(resourceName: str) -> StringIO:
 84    """
 85    Fetch a file-like resource from the `PYTHONPATH` and `resources` module
 86    included in the SSScore package.  Common resources include the drop zones
 87    list CSV and in-line documentation Markdown text.
 88
 89    Arguments
 90    ---------
 91        resourceName
 92    A string representing the resource file name, usually a CSV file.
 93
 94    Returns
 95    -------
 96    An instance of `StringIO` ready to be process as a text stream by the
 97    caller.
 98
 99    Raises
100    ------
101    `SSScoringError` if the resource dataframe isn't the global drop zones
102    directory or the file is invalid in any way.
103    """
104    try:
105        return StringIO(files(RESOURCES).joinpath(resourceName).read_bytes().decode(FLYSIGHT_FILE_ENCODING))
106    except Exception as e:
107        raise SSScoringError('Invalid resource - %s' % str(e))

110def initDropZonesFromResource() -> pd.DataFrame:
111    """
112    Get the DZs directory from a CSV enclosed in the distribution package as a
113    resource.  The resources package is fixed to `ssscoring.resources`, the
114    default resource file is defined by `DZ_DIRECTORY` but can be anything.
115
116    Returns
117    -------
118    The global drop zones directory as a dataframe.
119
120    Raises
121    ------
122    `SSScoringError` if the resource dataframe isn't the global drop zones
123    directory or the file is invalid in any way.
124    """
125    try:
126        buffer = fetchResource(DZ_DIRECTORY)
127        dropZones = pd.read_csv(buffer, sep=',')
128    except Exception as e:
129        raise SSScoringError('Invalid resource - %s' % str(e))
130
131    if 'dropZone' not in dropZones.columns:
132        raise SSScoringError('dropZone object is a dataframe but not the drop zones directory')
133
134    return dropZones

137def displayJumpDataIn(resultsTable: pd.DataFrame):
138    """
139    Display the individual results, as a table.
140
141    Arguments
142    ---------
143        resultsTable: pd.DataFrame
144    The results from a speed skydiving jump.
145
146    See
147    ---
148    `ssscoring.datatypes.JumpResults`
149    """
150    if resultsTable is not None:
151        table = resultsTable.copy()
152        table.vKMh = table.vKMh.apply(lambda x: round(x, 2))
153        table.hKMh = table.hKMh.apply(lambda x: round(x, 2))
154        table.deltaV = table.deltaV.apply(lambda x: round(x, 2))
155        table.deltaAngle = table.deltaAngle.apply(lambda x: round(x, 2))
156        table['altitude (ft)'] = table['altitude (ft)'].apply(lambda x: round(x, 1))
157        table.index = ['']*len(table)
158        st.dataframe(table, hide_index=True)

161def interpretJumpResult(tag: str,
162                        jumpResult: JumpResults,
163                        processBadJump: bool):
164    """
165    Interpret the jump results and generate the corresponding labels and
166    warnings if a jump is invalid, or only "somewhat valid" according to ISC
167    rules.  The caller turns results display on/off depending on training vs
168    in-competition settins.  Because heuristics are a beautiful thing.
169
170    Arguments
171    ---------
172        tag
173    A string that identifies a specific jump and the FlySight version that
174    generated the corresponding track file.  Often in the form: `HH-mm-ss:vX`
175    where `X` is the FlySight hardware version.
176
177        jumpResult
178    An instance of `ssscoring.datatypes.JumpResults` with jump data.
179
180        processBadJump
181    If `True`, generate end-user warnings as part of its results processing if
182    the jump is invalid because of ISC rules, but set the status to OK so that
183    the jump may be displayed.
184
185    Returns
186    -------
187    A `tuple` of these objects:
188
189    - `jumpStatusInfo` - the jump status, in human-readable form
190    - `scoringInfo` - Max speed, scoring window, etc.
191    - `badJumpLegend` - A warning or error if the jump is invalid according to
192      ISC rules
193    - `jumpStatus` - An instance of `JumpStatus` that may have been overriden to
194      `OK` if `processBadJump` was set to `True` and the jump was invalid.  Used
195      only for display.  Use `jumpResult.status` to determine the actual result
196      of the jump from a strict scoring perspective.  `jumpStatus` is  used
197      for display override purposes only.
198    """
199    maxSpeed = jumpResult.maxSpeed
200    window = jumpResult.window
201    jumpStatus = jumpResult.status
202    jumpStatusInfo = ''
203    if jumpResult.status == JumpStatus.WARM_UP_FILE:
204        badJumpLegend = '<span style="color: red">Warm up file or SMD ran out of battery - nothing to do<br>'
205        scoringInfo = ''
206    elif jumpResult.status == JumpStatus.SPEED_ACCURACY_EXCEEDS_LIMIT:
207        badJumpLegend = '<span style="color: red">%s - RE-JUMP: speed accuracy exceeds ISC threshold<br>' % tag
208        scoringInfo = ''
209    elif jumpResult.status == JumpStatus.INVALID_SPEED_FILE:
210        badJumpLegend = '<span style="color: red">Invalid or corrupted FlySight file - it\'s neither version 1 nor version 2<br>'
211        scoringInfo = ''
212    else:
213        scoringInfo = 'Max speed = {0:,.0f}; '.format(maxSpeed)+('exit at %d m (%d ft)<br>Validation window starts at %d m (%d ft)<br>End scoring window at %d m (%d ft)<br>' % \
214                        (window.start, M_2_FT*window.start, window.validationStart, M_2_FT*window.validationStart, window.end, M_2_FT*window.end))
215    if (processBadJump and jumpStatus != JumpStatus.OK and jumpStatus != JumpStatus.WARM_UP_FILE and jumpStatus != JumpStatus.SPEED_ACCURACY_EXCEEDS_LIMIT and jumpStatus != JumpStatus.INVALID_SPEED_FILE) or jumpStatus == JumpStatus.OK:
216        jumpStatusInfo = '<span style="color: %s">%s jump - %s - %.02f km/h</span><br>' % ('green', tag, 'VALID', jumpResult.score)
217        belowMaxAltitude = isValidMaximumAltitude(jumpResult.data.altitudeAGL.max())
218        badJumpLegend = None
219        if not isValidMinimumAltitude(jumpResult.data.altitudeAGL.max()):
220            badJumpLegend = '<span style="color: yellow"><span style="font-weight: bold">Warning:</span> exit altitude AGL was lower than the minimum scoring altitude<br>'
221            jumpStatus = JumpStatus.ALTITUDE_EXCEEDS_MINIMUM
222        if not belowMaxAltitude:
223            jumpStatusInfo = '<span style="color: %s">%s jump - %s - %.02f km/h</span><br>' % ('red', tag, 'INVALID', jumpResult.score)
224            badJumpLegend = '<span style="color: red"><span style="font-weight: bold">RE-JUMP:</span> exit altitude AGL exceeds the maximum altitude<br>'
225            jumpStatus = JumpStatus.ALTITUDE_EXCEEDS_MAXIMUM
226    return jumpStatusInfo, scoringInfo, badJumpLegend, jumpStatus

282def initFileUploaderState(filesObject:str, uploaderKey:str ='uploaderKey'):
283    """
284    Initialize the session state for the Streamlit app uploader so that
285    selections can be cleared in callbacks later.
286
287    **Important**: `initFileUploaderState()` __must__ be called after setting the
288    page configuration (per Streamlit architecture rules) and before adding any
289    widgets to the sidebars, containers, or main application.
290
291    Argument
292    --------
293        filesObject
294    A `str` name that is either `'trackFile'` for the single track file process
295    or `'trackFiles'` for the app page that handles more than one track file at
296    a time.
297
298        uploaderKey
299    A unique identifier for the uploader key component, usually set to
300    `'uploaderKey'` but can be any arbitrary name.  This value must match the
301    `file_uploader(..., key=uploaderKey,...)` value.
302
303    """
304    if filesObject not in st.session_state:
305        st.session_state[filesObject] = None
306    if uploaderKey not in st.session_state:
307        st.session_state[uploaderKey] = 0

329@st.dialog('DZ Coordinates')
330def displayDZCoordinates():
331    """
332    Display the DZ coordinates in a dialog if one is selected in the drop zones
333    selection box.  The selection is stored in the  `st.session_state.currentDropZone`
334    variable.  If a valid name is available, the latitude and longitude are
335    displayed.  If `None` or invalid, a notification dialog is displayed.
336
337    The corresponding UI button is only enabled if the user selected a valid DZ,
338    otherwise the button is disabled.
339    """
340    dropZones = initDropZonesFromResource()
341    currentDZ = dropZones[dropZones.dropZone == st.session_state.currentDropZone ].iloc[0]
342    lat = float(currentDZ.lat)
343    lon = float(currentDZ.lon)
344    st.write(st.session_state.currentDropZone)
345    st.write('%.4f, %.4f - elevation: %.2f m, %.2f ft MSL' % (lat, lon, currentDZ.elevation, currentDZ.elevation*M_2_FT))
346    if st.button('OK'):
347        st.rerun()

350def setSideBarAndMain(icon: str, singleTrack: bool, selectDZState):
351    """
352    Set all the interactive and navigational components for the app's side bar.
353
354    Arguments
355    ---------
356        icon
357    A meaningful Emoji associated with the the side bar's title.
358
359        singleTrack
360    A flag for allowing selection of a single or multiple track files in the
361    corresponding selector component.  Determines whether the side bar is used
362    for the single- or multiple selection application.
363
364        selectDZState
365    A callback for the drop zone selector selection box, affected by events in
366    the main application.
367
368    Notes
369    -----
370    All the aplication level values associated with the components and
371    selections from the side bar are stored in `st.session_state` and visible
372    across the whole application.
373
374    **Do not** cache calls to `ssscoring.appcommon.setSideBarAndMain()` because
375    this can result in unpredictable behavior since the cache may never be
376    cleared until application reload.
377    """
378    dropZones = initDropZonesFromResource()
379    st.session_state.currentDropZone = None
380    elevation = None
381    st.sidebar.title('%s SSScore %s' % (icon, __VERSION__))
382    st.session_state.processBadJump = st.sidebar.checkbox('Process bad jumps', value=True, help='Display results from invalid jumps')
383    st.session_state.currentDropZone = st.sidebar.selectbox('Select the drop zone:', dropZones.dropZone, index=None, on_change=selectDZState, disabled=(elevation != None and elevation != 0.0))
384    elevation = st.sidebar.number_input('...or enter the DZ elevation in meters:', min_value=0.0, max_value=4000.0, value='min', format='%.2f', disabled=(st.session_state.currentDropZone != None), on_change=selectDZState)
385    if st.session_state.currentDropZone:
386        st.session_state.elevation = dropZones[dropZones.dropZone == st.session_state.currentDropZone ].iloc[0].elevation
387    elif elevation != None and elevation != 0.0:
388        st.session_state.elevation= elevation
389    else:
390        st.session_state.elevation = None
391        st.session_state.trackFiles = None
392    st.sidebar.metric('Elevation', value='%.1f m' % (0.0 if st.session_state.elevation == None else st.session_state.elevation))
393    if singleTrack:
394        trackFile = st.sidebar.file_uploader('Track file', type=[ 'csv', ], disabled=st.session_state.elevation == None, key = st.session_state.uploaderKey)
395        if trackFile:
396            st.session_state.trackFile = trackFile
397    else:
398        trackFiles = st.sidebar.file_uploader(
399            'Track files',
400            type=[ 'csv', ],
401            disabled=st.session_state.elevation == None,
402            accept_multiple_files=True,
403            key = st.session_state.uploaderKey
404        )
405        if trackFiles:
406            st.session_state.trackFiles = trackFiles
407    st.sidebar.button('Clear', on_click=selectDZState)
408    st.sidebar.button('Display DZ coordinates', on_click=displayDZCoordinates, disabled=(st.session_state.currentDropZone == None))
409    st.sidebar.link_button('Report missing DZ', 'https://github.com/pr3d4t0r/SSScoring/issues/new?template=report-missing-dz.md', icon=':material/breaking_news_alt_1:')
410    st.sidebar.link_button('Feature request or bug report', 'https://github.com/pr3d4t0r/SSScoring/issues/new?template=Blank+issue', icon=':material/breaking_news_alt_1:')

163                        jumpResult: JumpResults,
164                        processBadJump: bool):
165    """
166    Interpret the jump results and generate the corresponding labels and
167    warnings if a jump is invalid, or only "somewhat valid" according to ISC
168    rules.  The caller turns results display on/off depending on training vs
169    in-competition settins.  Because heuristics are a beautiful thing.
170
171    Arguments
172    ---------
173        tag
174    A string that identifies a specific jump and the FlySight version that
175    generated the corresponding track file.  Often in the form: `HH-mm-ss:vX`
176    where `X` is the FlySight hardware version.
177
178        jumpResult
179    An instance of `ssscoring.datatypes.JumpResults` with jump data.
180
181        processBadJump
182    If `True`, generate end-user warnings as part of its results processing if
183    the jump is invalid because of ISC rules, but set the status to OK so that
184    the jump may be displayed.
185
186    Returns
187    -------
188    A `tuple` of these objects:
189
190    - `jumpStatusInfo` - the jump status, in human-readable form
191    - `scoringInfo` - Max speed, scoring window, etc.
192    - `badJumpLegend` - A warning or error if the jump is invalid according to
193      ISC rules
194    - `jumpStatus` - An instance of `JumpStatus` that may have been overriden to
195      `OK` if `processBadJump` was set to `True` and the jump was invalid.  Used
196      only for display.  Use `jumpResult.status` to determine the actual result
197      of the jump from a strict scoring perspective.  `jumpStatus` is  used
198      for display override purposes only.
199    """
200    maxSpeed = jumpResult.maxSpeed
201    window = jumpResult.window
202    jumpStatus = jumpResult.status
203    jumpStatusInfo = ''
204    if jumpResult.status == JumpStatus.WARM_UP_FILE:
205        badJumpLegend = '<span style="color: red">Warm up file or SMD ran out of battery - nothing to do<br>'
206        scoringInfo = ''
207    elif jumpResult.status == JumpStatus.SPEED_ACCURACY_EXCEEDS_LIMIT:
208        badJumpLegend = '<span style="color: red">%s - RE-JUMP: speed accuracy exceeds ISC threshold<br>' % tag
209        scoringInfo = ''
210    elif jumpResult.status == JumpStatus.INVALID_SPEED_FILE:
211        badJumpLegend = '<span style="color: red">Invalid or corrupted FlySight file - it\'s neither version 1 nor version 2<br>'
212        scoringInfo = ''
213    else:
214        scoringInfo = 'Max speed = {0:,.0f}; '.format(maxSpeed)+('exit at %d m (%d ft)<br>Validation window starts at %d m (%d ft)<br>End scoring window at %d m (%d ft)<br>' % \
215                        (window.start, M_2_FT*window.start, window.validationStart, M_2_FT*window.validationStart, window.end, M_2_FT*window.end))
216    if (processBadJump and jumpStatus != JumpStatus.OK and jumpStatus != JumpStatus.WARM_UP_FILE and jumpStatus != JumpStatus.SPEED_ACCURACY_EXCEEDS_LIMIT and jumpStatus != JumpStatus.INVALID_SPEED_FILE) or jumpStatus == JumpStatus.OK:
217        jumpStatusInfo = '<span style="color: %s">%s jump - %s - %.02f km/h</span><br>' % ('green', tag, 'VALID', jumpResult.score)
218        belowMaxAltitude = isValidMaximumAltitude(jumpResult.data.altitudeAGL.max())
219        badJumpLegend = None
220        if not isValidMinimumAltitude(jumpResult.data.altitudeAGL.max()):
221            badJumpLegend = '<span style="color: yellow"><span style="font-weight: bold">Warning:</span> exit altitude AGL was lower than the minimum scoring altitude<br>'
222            jumpStatus = JumpStatus.ALTITUDE_EXCEEDS_MINIMUM
223        if not belowMaxAltitude:
224            jumpStatusInfo = '<span style="color: %s">%s jump - %s - %.02f km/h</span><br>' % ('red', tag, 'INVALID', jumpResult.score)
225            badJumpLegend = '<span style="color: red"><span style="font-weight: bold">RE-JUMP:</span> exit altitude AGL exceeds the maximum altitude<br>'
226            jumpStatus = JumpStatus.ALTITUDE_EXCEEDS_MAXIMUM
227    return jumpStatusInfo, scoringInfo, badJumpLegend, jumpStatus
